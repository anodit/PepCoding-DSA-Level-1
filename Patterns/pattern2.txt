import java.util.*;

public class Main {

    public static void main(String[] args) {
        Scanner scn = new Scanner(System.in);
        int n = scn.nextInt();
        int i,j;
         for(i=1;i<=n;i++)
        {
            for(j=1;j<=(n-i+1);j++)
            {
                System.out.print("*	");
            }
            System.out.println();
        }

    }
}
