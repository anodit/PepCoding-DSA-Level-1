import java.io.*;
import java.util.*;

public class Main {

    public static void main(String[] args) throws Exception {
        // write your code here
        Scanner scn = new Scanner(System.in);
        int x = scn.nextInt();
        int n = scn.nextInt();
        
        int pwr = power(x,n);
        System.out.println(pwr);
    }

    public static int power(int x, int n){
        if(n==0){
            return 1;
        }
        
        return x*power(x,n-1);
    }

}
